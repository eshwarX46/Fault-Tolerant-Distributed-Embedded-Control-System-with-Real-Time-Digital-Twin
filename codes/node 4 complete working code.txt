// ESP32_Node4_TFT_Working.ino
// Node 4 - Gateway + TFT Display (FIXED)

#include <CAN.h>
#include <TFT_eSPI.h>

// ============ CAN PIN DEFINITIONS ============
#define TX_GPIO_NUM   21
#define RX_GPIO_NUM   22

// ============ TFT OBJECT ============
TFT_eSPI tft = TFT_eSPI();

// ============ NODE STATUS STRUCTURES ============
struct Node1Status {
  bool active = false;
  unsigned long lastDataReceived = 0;
  bool fault = false;
  float temp1 = 0;
  float temp2 = 0;
  float current = 0;
  int16_t accelX = 0;
};

struct Node2Status {
  bool active = false;
  unsigned long lastDataReceived = 0;
  bool fault = false;
  int motorSpeed = 0;
  bool motorDirection = true;
  bool relay1State = false;
  bool relay2State = false;
};

struct Node3Status {
  bool active = false;
  unsigned long lastHeartbeat = 0;
  bool fault = false;
};

Node1Status node1;
Node2Status node2;
Node3Status node3;

unsigned long lastDisplayUpdate = 0;

// ============ COLORS ============
#define COLOR_BG     TFT_BLACK
#define COLOR_OK     TFT_GREEN
#define COLOR_FAULT  TFT_RED
#define COLOR_WARN   TFT_YELLOW
#define COLOR_TEXT   TFT_WHITE
#define COLOR_HEADER TFT_CYAN

void setup() {
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("=================================");
  Serial.println("ESP32 - NODE 4 (GATEWAY + TFT)");
  Serial.println("=================================");
  
  Serial.print("Initializing TFT... ");
  tft.init();
  tft.setRotation(1);
  tft.fillScreen(COLOR_BG);
  tft.setTextColor(COLOR_TEXT);
  tft.setTextSize(2);
  tft.drawString("Initializing...", 80, 150);
  Serial.println("OK ✓");
  delay(1000);
  
  Serial.print("Initializing CAN... ");
  CAN.setPins(RX_GPIO_NUM, TX_GPIO_NUM);
  
  if (!CAN.begin(250E3)) {
    Serial.println("FAILED!");
    tft.fillScreen(COLOR_FAULT);
    tft.drawString("CAN ERROR!", 80, 150);
    while (1);
  }
  Serial.println("OK ✓");
  
  Serial.println("✅ CAN Ready at 250KBPS!");
  Serial.println("Waiting for Node 1, Node 2 and Node 3 data...");
  
  drawDisplay();
}

void loop() {
  unsigned long now = millis();
  
  receiveCAN();
  
  static unsigned long lastCheck = 0;
  if (now - lastCheck > 1000) {
    checkTimeouts();
    lastCheck = now;
  }
  
  if (now - lastDisplayUpdate > 200) {
    drawDisplay();
    lastDisplayUpdate = now;
  }
  
  static unsigned long lastHeartbeat = 0;
  if (now - lastHeartbeat > 2000) {
    sendHeartbeat();
    lastHeartbeat = now;
  }
  
  delay(10);
}

void receiveCAN() {
  int packetSize = CAN.parsePacket();
  
  if (packetSize) {
    byte data[8];
    int i = 0;
    while (CAN.available() && i < 8) {
      data[i++] = CAN.read();
    }
    
    // Node 1 Sensor Data (ID: 0x101)
    if (CAN.packetId() == 0x101) {
      node1.lastDataReceived = millis();
      node1.active = true;
      
      if (node1.fault) {
        node1.fault = false;
        Serial.println("✅ Node 1 RECOVERED!");
      }
      
      if (i >= 8) {
        node1.temp1 = data[3];
        node1.temp2 = data[4];
        node1.current = data[5] / 100.0;
        node1.accelX = (data[6] << 8) | data[7];
      }
      
      Serial.print("Node1 - Temp1:");
      Serial.print(node1.temp1);
      Serial.print("C, Current:");
      Serial.println(node1.current);
    }
    
    // Node 2 Status Data (ID: 0x200)
    else if (CAN.packetId() == 0x200) {
      node2.lastDataReceived = millis();
      node2.active = true;
      
      if (node2.fault) {
        node2.fault = false;
        Serial.println("✅ Node 2 RECOVERED!");
      }
      
      if (i >= 8) {
        node2.motorSpeed = data[3];
        node2.motorDirection = (data[4] == 1);
        node2.relay1State = (data[5] == 1);
        node2.relay2State = (data[6] == 1);
      }
      
      Serial.print("Node2 - Motor:");
      Serial.print(node2.motorSpeed);
      Serial.println(" RPM");
    }
    
    // Node 3 Heartbeat (ID: 0x303)
    else if (CAN.packetId() == 0x303) {
      node3.lastHeartbeat = millis();
      node3.active = true;
      
      if (node3.fault) {
        node3.fault = false;
        Serial.println("✅ Node 3 RECOVERED!");
      }
      
      Serial.println("💓 Node 3 Heartbeat received");
    }
  }
}

void checkTimeouts() {
  unsigned long now = millis();
  
  // Node 1 timeout (4 seconds)
  if (node1.active && (now - node1.lastDataReceived > 4000)) {
    if (!node1.fault) {
      node1.fault = true;
      Serial.println("🚨 FAULT! NODE 1 NOT RESPONDING!");
    }
  }
  
  // Node 2 timeout (4 seconds)
  if (node2.active && (now - node2.lastDataReceived > 4000)) {
    if (!node2.fault) {
      node2.fault = true;
      Serial.println("🚨 FAULT! NODE 2 NOT RESPONDING!");
    }
  }
  
  // Node 3 timeout (4 seconds)
  if (node3.active && (now - node3.lastHeartbeat > 4000)) {
    if (!node3.fault) {
      node3.fault = true;
      Serial.println("🚨 FAULT! NODE 3 NOT RESPONDING!");
    }
  }
}

void drawDisplay() {
  tft.fillScreen(COLOR_BG);
  
  // ============ HEADER ============
  tft.fillRect(0, 0, 320, 25, COLOR_HEADER);
  tft.setTextColor(TFT_BLACK, COLOR_HEADER);
  tft.setTextSize(1);
  tft.drawString("FAULT-TOLERANT SYSTEM", 160, 8);
  
  int yOffset = 35;
  int lineHeight = 25;
  
  // ============ NODE 1 SECTION ============
  tft.setTextColor(COLOR_TEXT, COLOR_BG);
  tft.setTextSize(1);
  tft.drawString("NODE 1 - SENSOR HUB", 10, yOffset);
  
  uint16_t node1Color = node1.fault ? COLOR_FAULT : (node1.active ? COLOR_OK : COLOR_WARN);
  tft.fillCircle(280, yOffset + 5, 5, node1Color);
  
  yOffset += lineHeight;
  tft.drawString("Temp #1: " + String(node1.temp1, 1) + "C", 10, yOffset);
  tft.drawString("Temp #2: " + String(node1.temp2, 1) + "C", 160, yOffset);
  
  yOffset += lineHeight;
  tft.drawString("Current: " + String(node1.current, 2) + "A", 10, yOffset);
  tft.drawString("Accel X: " + String(node1.accelX), 160, yOffset);
  
  if (node1.fault) {
    yOffset += lineHeight;
    tft.setTextColor(TFT_YELLOW, COLOR_BG);
    tft.drawString("!!! NODE 1 OFFLINE !!!", 10, yOffset);
    tft.setTextColor(COLOR_TEXT, COLOR_BG);
  }
  
  // ============ NODE 2 SECTION ============
  yOffset += lineHeight + 5;
  tft.drawString("NODE 2 - ACTUATOR", 10, yOffset);
  
  uint16_t node2Color = node2.fault ? COLOR_FAULT : (node2.active ? COLOR_OK : COLOR_WARN);
  tft.fillCircle(280, yOffset + 5, 5, node2Color);
  
  yOffset += lineHeight;
  tft.drawString("Motor Speed: " + String(node2.motorSpeed) + " RPM", 10, yOffset);
  tft.drawString("Direction: " + String(node2.motorDirection ? "FWD" : "REV"), 160, yOffset);
  
  yOffset += lineHeight;
  tft.drawString("Relay 1: " + String(node2.relay1State ? "ON" : "OFF"), 10, yOffset);
  tft.drawString("Relay 2: " + String(node2.relay2State ? "ON" : "OFF"), 160, yOffset);
  
  if (node2.fault) {
    yOffset += lineHeight;
    tft.setTextColor(TFT_YELLOW, COLOR_BG);
    tft.drawString("!!! NODE 2 OFFLINE !!!", 10, yOffset);
    tft.setTextColor(COLOR_TEXT, COLOR_BG);
  }
  
  // ============ FOOTER ============
  yOffset += lineHeight + 7;
  
  // System Fault triggers if ANY node (1, 2, or 3) has fault
  if (node1.fault || node2.fault || node3.fault) {
    tft.fillRect(5, yOffset, 310, 20, COLOR_FAULT);
    tft.setTextColor(TFT_YELLOW, COLOR_FAULT);
    tft.drawString("SYSTEM FAULT!", 160, yOffset + 7);
  } else if (node1.active && node2.active && node3.active) {
    tft.fillRect(5, yOffset, 310, 20, COLOR_OK);
    tft.setTextColor(TFT_BLACK, COLOR_OK);
    tft.drawString("SYSTEM NORMAL", 160, yOffset + 7);
  } else {
    tft.fillRect(5, yOffset, 310, 20, COLOR_WARN);
    tft.setTextColor(TFT_BLACK, COLOR_WARN);
    tft.drawString("WAITING FOR NODES...", 160, yOffset + 7);
  }
}

void sendHeartbeat() {
  CAN.beginPacket(0x304);
  CAN.write(4);
  CAN.write(99);
  CAN.write(millis() / 1000);
  CAN.write('O');
  CAN.write('K');
  CAN.endPacket();
}
