// ESP32_Node3_Monitor_And_Control.ino
// ESP32 as Node 3 - Monitors BOTH Node 1 AND Node 2 + Controls Node 2

#include <CAN.h>

// ============ CAN PIN DEFINITIONS ============
#define TX_GPIO_NUM   21
#define RX_GPIO_NUM   22

// ============ LED & BUZZER & BUTTON PIN DEFINITIONS ============
#define GREEN_LED   2     // Green LED - Node 1 status indicator
#define WHITE_LED   4     // White LED - Node 2 status indicator
#define BLUE_LED    15    // Blue LED - Fault indicator
#define BUZZER      16    // Active Buzzer
#define RESET_BTN   17    // Push Button (Reset)

// ============ NODE 1 STATUS ============
struct Node1Status {
  bool active;
  unsigned long lastDataReceived;
  int missedCount;
  bool fault;
  float temp1;
  float temp2;
  float current;
  int16_t accelX;
};

Node1Status node1;

// ============ NODE 2 STATUS ============
struct Node2Status {
  bool active;
  unsigned long lastDataReceived;
  int missedCount;
  bool fault;
  int motorSpeed;
  bool motorDirection;
  bool relay1State;
  bool relay2State;
  bool emergencyStop;
};

Node2Status node2;

// ============ CONTROL VARIABLES ============
int sendCount = 0;
int commandIndex = 0;
unsigned long lastSendTime = 0;
unsigned long lastHeartbeatCheck = 0;
unsigned long lastCommandTime = 0;
int buttonPressCount = 0;
int sensorDataCount = 0;
int statusDataCount = 0;
bool emergencyStopActive = false;

// ============ LED BLINKING VARIABLES ============
unsigned long lastFaultBlink = 0;
bool faultBlinkState = false;

void setup() {
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("=================================");
  Serial.println("ESP32 - NODE 3 (MONITOR + CONTROL)");
  Serial.println("Monitoring Node 1 AND Node 2 + Controlling Node 2");
  Serial.println("=================================");
  
  // ============ INITIALIZE LEDS ============
  pinMode(GREEN_LED, OUTPUT);
  pinMode(WHITE_LED, OUTPUT);
  pinMode(BLUE_LED, OUTPUT);
  pinMode(BUZZER, OUTPUT);
  pinMode(RESET_BTN, INPUT_PULLUP);
  
  digitalWrite(GREEN_LED, LOW);
  digitalWrite(WHITE_LED, LOW);
  digitalWrite(BLUE_LED, LOW);
  digitalWrite(BUZZER, LOW);
  
  startupTest();
  
  // ============ INITIALIZE NODE 1 STATUS ============
  node1.active = false;
  node1.lastDataReceived = 0;
  node1.missedCount = 0;
  node1.fault = false;
  
  // ============ INITIALIZE NODE 2 STATUS ============
  node2.active = false;
  node2.lastDataReceived = 0;
  node2.missedCount = 0;
  node2.fault = false;
  node2.motorSpeed = 0;
  node2.motorDirection = true;
  node2.relay1State = false;
  node2.relay2State = false;
  node2.emergencyStop = false;
  
  // ============ INITIALIZE CAN ============
  CAN.setPins(RX_GPIO_NUM, TX_GPIO_NUM);
  
  if (!CAN.begin(250E3)) {
    Serial.println("❌ Starting CAN failed!");
    while (1);
  }
  
  Serial.println("✅ CAN Ready at 250KBPS!");
  Serial.println("");
  Serial.println("Monitoring Node 1 (CAN ID: 0x101) and Node 2 (CAN ID: 0x200)...");
  Serial.println("Sending control commands to Node 2 every 10 seconds...");
  Serial.println("=================================\n");
}

void loop() {
  unsigned long now = millis();
  
  // ============ CHECK RESET BUTTON ============
  if (digitalRead(RESET_BTN) == LOW) {
    buttonPressCount++;
    Serial.print("🔘 RESET BUTTON PRESSED! (");
    Serial.print(buttonPressCount);
    Serial.println(" times)");
    
    digitalWrite(BUZZER, HIGH);
    delay(200);
    digitalWrite(BUZZER, LOW);
    clearAllFaults();
    delay(500);
  }
  
  // ============ SEND NODE 3 HEARTBEAT ============
  if (now - lastSendTime > 2000) {
    sendMyHeartbeat();
    lastSendTime = now;
  }
  
  // ============ SEND CONTROL COMMANDS TO NODE 2 ============
  if (!emergencyStopActive && (now - lastCommandTime > 10000)) {
    sendControlCommand();
    lastCommandTime = now;
  }
  
  // ============ CHECK NODE TIMEOUTS ============
  if (now - lastHeartbeatCheck > 1000) {
    checkNode1Timeout();
    checkNode2Timeout();
    lastHeartbeatCheck = now;
  }
  
  // ============ UPDATE LEDS ============
  updateLEDs();
  
  // ============ RECEIVE MESSAGES ============
  canReceiver();
  
  delay(10);
}

void startupTest() {
  Serial.println("Running startup test...");
  
  digitalWrite(GREEN_LED, HIGH);
  Serial.println("🟢 Green LED ON");
  delay(500);
  digitalWrite(GREEN_LED, LOW);
  
  digitalWrite(WHITE_LED, HIGH);
  Serial.println("⚪ White LED ON");
  delay(500);
  digitalWrite(WHITE_LED, LOW);
  
  digitalWrite(BLUE_LED, HIGH);
  Serial.println("🔵 Blue LED ON");
  delay(500);
  digitalWrite(BLUE_LED, LOW);
  
  digitalWrite(BUZZER, HIGH);
  Serial.println("🔊 Buzzer ON");
  delay(300);
  digitalWrite(BUZZER, LOW);
  Serial.println("🔇 Buzzer OFF");
  
  digitalWrite(GREEN_LED, HIGH);
  digitalWrite(WHITE_LED, HIGH);
  digitalWrite(BLUE_LED, HIGH);
  Serial.println("🟢⚪🔵 All LEDs ON");
  delay(500);
  digitalWrite(GREEN_LED, LOW);
  digitalWrite(WHITE_LED, LOW);
  digitalWrite(BLUE_LED, LOW);
  
  Serial.println("✅ Startup test complete!\n");
}

void sendMyHeartbeat() {
  CAN.beginPacket(0x303);
  CAN.write(3);
  CAN.write(sendCount);
  CAN.write(millis() / 1000);
  CAN.endPacket();
  sendCount++;
}

void sendControlCommand() {
  Serial.println("");
  Serial.println("=========================================");
  Serial.print(">>> Sending Control Command #");
  Serial.print(commandIndex + 1);
  Serial.println(" to NODE 2 <<<");
  
  switch(commandIndex) {
    case 0:
      Serial.println("Command: Set Motor Speed = 150");
      sendCommandToNode2(10, 150);
      break;
      
    case 1:
      Serial.println("Command: Set Motor Direction = FORWARD");
      sendCommandToNode2(12, 1);
      break;
      
    case 2:
      Serial.println("Command: Relay 1 = ON");
      sendCommandToNode2(13, 1);
      break;
      
    case 3:
      Serial.println("Command: Relay 2 = ON");
      sendCommandToNode2(14, 1);
      break;
      
    case 4:
      Serial.println("Command: Relay 1 = OFF");
      sendCommandToNode2(13, 0);
      break;
      
    case 5:
      Serial.println("Command: Relay 2 = OFF");
      sendCommandToNode2(14, 0);
      break;
      
    case 6:
      Serial.println("Command: Set Motor Speed = 100");
      sendCommandToNode2(10, 100);
      break;
      
    case 7:
      Serial.println("Command: Set Motor Direction = REVERSE");
      sendCommandToNode2(12, 0);
      break;
      
    case 8:
      Serial.println("Command: Set Motor Speed = 0 (STOP)");
      sendCommandToNode2(10, 0);
      break;
  }
  
  commandIndex++;
  if (commandIndex > 8) {
    commandIndex = 0;
  }
  
  Serial.println(">>> COMMAND SENT <<<");
  Serial.println("=========================================\n");
}

void sendCommandToNode2(int command, int value) {
  CAN.beginPacket(0x401);
  CAN.write(2);     // Target Node 2
  CAN.write(command);
  CAN.write(value);
  CAN.endPacket();
}

void canReceiver() {
  int packetSize = CAN.parsePacket();
  
  if (packetSize) {
    byte data[8];
    int i = 0;
    while (CAN.available() && i < 8) {
      data[i++] = CAN.read();
    }
    
    // ============ NODE 1 SENSOR DATA (ID: 0x101) ============
    if (CAN.packetId() == 0x101) {
      sensorDataCount++;
      node1.lastDataReceived = millis();
      node1.active = true;
      node1.missedCount = 0;
      
      if (node1.fault) {
        node1.fault = false;
        Serial.println("✅ Node 1 has RECOVERED!");
        checkAndClearBuzzer();
      }
      
      if (i >= 8) {
        node1.temp1 = data[3];
        node1.temp2 = data[4];
        node1.current = data[5] / 100.0;
        node1.accelX = (data[6] << 8) | data[7];
        
        Serial.println("");
        Serial.println("=========================================");
        Serial.print("📥 NODE 1 DATA #");
        Serial.println(sensorDataCount);
        Serial.println("=========================================");
        Serial.print("🌡️ Temperature #1: ");
        Serial.print(node1.temp1);
        Serial.println(" °C");
        Serial.print("🌡️ Temperature #2: ");
        Serial.print(node1.temp2);
        Serial.println(" °C");
        Serial.print("⚡ Current: ");
        Serial.print(node1.current);
        Serial.println(" A");
        Serial.print("📳 Accel X: ");
        Serial.println(node1.accelX);
        Serial.println("=========================================");
        Serial.println("");
      }
    }
    
    // ============ NODE 2 STATUS DATA (ID: 0x200) ============
    else if (CAN.packetId() == 0x200) {
      statusDataCount++;
      node2.lastDataReceived = millis();
      node2.active = true;
      node2.missedCount = 0;
      
      if (node2.fault) {
        node2.fault = false;
        Serial.println("✅ Node 2 has RECOVERED!");
        checkAndClearBuzzer();
      }
      
      if (i >= 8) {
        node2.motorSpeed = data[3];
        node2.motorDirection = (data[4] == 1);
        node2.relay1State = (data[5] == 1);
        node2.relay2State = (data[6] == 1);
        node2.emergencyStop = (data[7] == 1);
        
        Serial.println("");
        Serial.println("=========================================");
        Serial.print("📥 NODE 2 STATUS #");
        Serial.println(statusDataCount);
        Serial.println("=========================================");
        Serial.print("🎯 Motor Speed: ");
        Serial.println(node2.motorSpeed);
        Serial.print("🔄 Motor Direction: ");
        Serial.println(node2.motorDirection ? "FORWARD" : "REVERSE");
        Serial.print("🔴 Relay 1: ");
        Serial.println(node2.relay1State ? "ON" : "OFF");
        Serial.print("🔴 Relay 2: ");
        Serial.println(node2.relay2State ? "ON" : "OFF");
        Serial.print("🚨 Emergency Stop: ");
        Serial.println(node2.emergencyStop ? "ACTIVE" : "INACTIVE");
        Serial.println("=========================================");
        Serial.println("");
      }
    }
  }
}

void checkNode1Timeout() {
  unsigned long now = millis();
  
  if (node1.active && (now - node1.lastDataReceived > 4000)) {
    node1.missedCount++;
    
    if (node1.missedCount >= 1 && !node1.fault) {
      node1.fault = true;
      Serial.println("");
      Serial.println("=========================================");
      Serial.println("🚨 FAULT DETECTED! NODE 1 NOT RESPONDING!");
      Serial.println("=========================================");
      Serial.println("");
      
      digitalWrite(BUZZER, HIGH);
    }
  }
}

void checkNode2Timeout() {
  unsigned long now = millis();
  
  if (node2.active && (now - node2.lastDataReceived > 4000)) {
    node2.missedCount++;
    
    if (node2.missedCount >= 1 && !node2.fault) {
      node2.fault = true;
      Serial.println("");
      Serial.println("=========================================");
      Serial.println("🚨 FAULT DETECTED! NODE 2 NOT RESPONDING!");
      Serial.println("=========================================");
      Serial.println("");
      
      digitalWrite(BUZZER, HIGH);
    }
  }
}

void updateLEDs() {
  // Green LED - Node 1 status
  digitalWrite(GREEN_LED, node1.active && !node1.fault);
  
  // White LED - Node 2 status
  digitalWrite(WHITE_LED, node2.active && !node2.fault);
  
  // Blue LED - Blinking when ANY fault detected
  if (node1.fault || node2.fault) {
    if (millis() - lastFaultBlink > 500) {
      faultBlinkState = !faultBlinkState;
      digitalWrite(BLUE_LED, faultBlinkState);
      lastFaultBlink = millis();
    }
  } else {
    digitalWrite(BLUE_LED, LOW);
    faultBlinkState = false;
  }
}

void checkAndClearBuzzer() {
  // Turn off buzzer if no faults
  if (!node1.fault && !node2.fault) {
    digitalWrite(BUZZER, LOW);
  }
}

void clearAllFaults() {
  node1.fault = false;
  node1.missedCount = 0;
  node1.active = false;
  
  node2.fault = false;
  node2.missedCount = 0;
  node2.active = false;
  
  digitalWrite(BLUE_LED, LOW);
  digitalWrite(BUZZER, LOW);
  faultBlinkState = false;
  emergencyStopActive = false;
  
  Serial.println("✅ All faults cleared! Waiting for Node 1 and Node 2 data...");
}
